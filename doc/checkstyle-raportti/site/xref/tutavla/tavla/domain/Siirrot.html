6   package tutavla.tavla.domain;

8   import java.util.*;
9   
10  /**
11   *
12   * @author ttuotila
13   */
14  public class Siirrot {
15  
16      private ArrayList<Noppa> nopat;
17      private ArrayList<Integer> siirrot;
18      private Random random;
19  
20      public Siirrot(Random random) {
21          siirrot = new ArrayList<>();
22          nopat = new ArrayList<>();
23          random = new Random();
24  
25          for (int i = 0; i < 2; i++) {
26              nopat.add(new Noppa(random));
27          }
28  
29      }
30  
31      /**
32       * Poistaa vanhat siirrot, heittää noppia uudestaan ja tallettaa uudet
33       * siirrot
34       */
35      public void heitaNopat() {
36  
37          siirrot.clear();
38  
39          for (int i = 0; i < 2; i++) {
40              nopat.get(i).heita();
41              if (siirrot.contains(nopat.get(i).getArvo())) {
42                  for (int j = 0; j < 3; j++) {
43                      siirrot.add(nopat.get(i).getArvo());
44                  }
45              } else {
46                  siirrot.add(nopat.get(i).getArvo());
47              }
48          }
49      }
50  
51      public ArrayList<Integer> haeSiirrot() {
52  
53          return siirrot;
54      }
55  
56  }
