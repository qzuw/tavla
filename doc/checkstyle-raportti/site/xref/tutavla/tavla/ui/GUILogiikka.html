267     /**
268      * Pelivuoro vaihtuu.
269      */
270     public void vuoroVaihtuu() {
271         if (sovelluslogiikka.onkoJokuVoittanut()) {
272             voitto();
273         }
274 
275         if (!onkoPeliKaynnissa && !lopetetaan) {
276             int noppa1 = sovelluslogiikka.haeSiirrot().get(0);
277             int noppa2 = sovelluslogiikka.haeSiirrot().get(1);
278             vuoroteksti = sovelluslogiikka.haeSiirtojarjestys().get(0) + " heitti " + noppa1 + " ja " + sovelluslogiikka.haeSiirtojarjestys().get(1) + " heitti " + noppa2 + ".";
279             if (noppa1 > noppa2) {
280                 sovelluslogiikka.pelaajaSiirtaaEnsin(true, 0);
281                 vuoroteksti += " " + sovelluslogiikka.haeVuorossaOlevaPelaaja().toString() + " liikkuu ensin.";
282                 this.onkoPeliKaynnissa = true;
283                 if (sovelluslogiikka.haeVuorossaOlevaPelaaja().onkoIhminen()) {
284                     this.pelaajaOnIhminen = true;
285                 } else {
286                     this.pelaajaOnIhminen = false;
287                 }
288             } else if (noppa1 < noppa2) {
289                 sovelluslogiikka.pelaajaSiirtaaEnsin(true, 1);
290                 vuoroteksti += " " + sovelluslogiikka.haeVuorossaOlevaPelaaja().toString() + " liikkuu ensin.";
291                 this.onkoPeliKaynnissa = true;
292                 if (sovelluslogiikka.haeVuorossaOlevaPelaaja().onkoIhminen()) {
293                     this.pelaajaOnIhminen = true;
294                 } else {
295                     this.pelaajaOnIhminen = false;
296                 }
297             } else {
298                 vuoroteksti += " Heitetään noppaa uudestaan.";
299                 sovelluslogiikka.heitaNopat();
300             }
301         } else if (!lopetetaan) {
302             vuoroteksti = "Vuoro vaihtuu.";
303 
304             sovelluslogiikka.vaihdaVuoroa();
305             sovelluslogiikka.nollaaLahtoruutu();
306             if (sovelluslogiikka.haeVuorossaOlevaPelaaja().onkoIhminen()) {
307                 this.pelaajaOnIhminen = true;
308             } else {
309                 this.pelaajaOnIhminen = false;
310             }
311             sovelluslogiikka.heitaNopat();
312         }
313         if (sovelluslogiikka.onkoJokuVoittanut()) {
314             voitto();
315         }
316 
317         paivitaSiirrotJaPelaaja();
318         kehys.revalidate();
319 
320     }

322     /**
323      * Pelaa näytä tietokonepelaajan vuoro.
324      */
325     public void pelaaTietokone() {
326         if (lopetetaan) {
327             lopeta();
328         }
329         while (!sovelluslogiikka.eiVoiSiirtaa() && sovelluslogiikka.haeSiirrot().size() > 0) {
330             sovelluslogiikka.pelaaTietokone();
331             paivitaSiirrotJaPelaaja();
332             kehys.revalidate();
333             piirtoalusta.repaint();
334         }
335         if (sovelluslogiikka.onkoJokuVoittanut()) {
336             voitto();
337         }
338         paivitaSiirrotJaPelaaja();
339         vuoroikkuna.setVisible(true);
340         kehys.revalidate();
341     }
