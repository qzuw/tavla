268     /**
269      * Pelivuoro vaihtuu.
270      */
271     public void vuoroVaihtuu() {
272         if (sovelluslogiikka.onkoJokuVoittanut()) {
273             voitto();
274         }
275 
276         if (!onkoPeliKaynnissa && !lopetetaan) {
277             arvoAloittaja();
278         } else if (!lopetetaan) {
279             vuoroteksti = "Vuoro vaihtuu.";
280 
281             sovelluslogiikka.vaihdaVuoroa();
282             sovelluslogiikka.nollaaLahtoruutu();
283 //            if (sovelluslogiikka.haeVuorossaOlevaPelaaja().onkoIhminen()) {
284 //                this.pelaajaOnIhminen = true;
285 //            } else {
286 //                this.pelaajaOnIhminen = false;
287 //            }
288             sovelluslogiikka.heitaNopat();
289         }
290         if (sovelluslogiikka.onkoJokuVoittanut()) {
291             voitto();
292         }
293 
294         paivitaSiirrotJaPelaaja();
295         kehys.revalidate();
296 
297     }

299     private void arvoAloittaja() {
300         int noppa1 = sovelluslogiikka.haeSiirrot().get(0);
301         int noppa2 = sovelluslogiikka.haeSiirrot().get(1);
302         vuoroteksti = sovelluslogiikka.haeSiirtojarjestys().get(0) + " heitti " + noppa1 + " ja " + sovelluslogiikka.haeSiirtojarjestys().get(1) + " heitti " + noppa2 + ".";
303         if (noppa1 > noppa2) {
304             kerroAloitusarvonnanVoittaja(0);
305         } else if (noppa1 < noppa2) {
306             kerroAloitusarvonnanVoittaja(1);
307         } else {
308             vuoroteksti += " Heitetään noppaa uudestaan.";
309             sovelluslogiikka.heitaNopat();
310         }
311     }
312 
313     private void kerroAloitusarvonnanVoittaja(int pelaaja) {
314         sovelluslogiikka.pelaajaSiirtaaEnsin(true, pelaaja);
315         vuoroteksti += " " + sovelluslogiikka.haeVuorossaOlevaPelaaja().toString() + " liikkuu ensin.";
316         this.onkoPeliKaynnissa = true;
317     }

319     /**
320      * Pelaa näytä tietokonepelaajan vuoro.
321      */
322     public void pelaaTietokone() {
323         if (lopetetaan) {
324             lopeta();
325         }
326         while (!sovelluslogiikka.eiVoiSiirtaa() && sovelluslogiikka.haeSiirrot().size() > 0) {
327             sovelluslogiikka.pelaaTietokone();
328             paivitaSiirrotJaPelaaja();
329             kehys.revalidate();
330             piirtoalusta.repaint();
331         }
332         if (sovelluslogiikka.onkoJokuVoittanut()) {
333             voitto();
334         }
335         paivitaSiirrotJaPelaaja();
336         vuoroikkuna.setVisible(true);
337         kehys.revalidate();
338     }
