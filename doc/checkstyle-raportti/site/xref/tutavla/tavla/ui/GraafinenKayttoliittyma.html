6   package tutavla.tavla.ui;
7   
8   import java.awt.BorderLayout;
9   import java.awt.Container;
10  import java.awt.Dimension;
11  import java.awt.GridLayout;
12  import javax.swing.JButton;
13  import javax.swing.JFrame;
14  import javax.swing.JLabel;
15  import javax.swing.JPanel;
16  import javax.swing.JTextField;
17  import javax.swing.WindowConstants;
18  import tutavla.tavla.logiikka.Sovelluslogiikka;
19  
20  /**
21   *
22   * @author ttuotila
23   */
24  public class GraafinenKayttoliittyma implements Runnable, Kayttoliittyma {
25      
26      private JFrame kehys;
27      private JFrame kyselyikkuna;
28      private Sovelluslogiikka svl;
29      private GUILogiikka guilg;
30      private JLabel siirrot;
31      private JLabel pelaaja;
32      private JLabel terveiset;
33      
34      public GraafinenKayttoliittyma() {
35          svl = new Sovelluslogiikka();
36          guilg = new GUILogiikka(svl);
37      }
38      
39      @Override
40      public void kaynnista() {
41          run();
42      }
43      
44      @Override
45      public void run() {
46          kehys = new JFrame("Tavla");
47          kehys.setPreferredSize(new Dimension(470, 425));
48          
49          kehys.setDefaultCloseOperation(WindowConstants.EXIT_ON_CLOSE);
50          
51          svl.heitaNopat();
52          
53          luoKomponentit(kehys.getContentPane());
54          
55          kehys.pack();
56          kehys.setVisible(true);
57          
58          kyselyikkuna = new JFrame("Tavla");
59          kyselyikkuna.setPreferredSize(new Dimension(250, 150));
60          
61          kyselyikkuna.setDefaultCloseOperation(WindowConstants.EXIT_ON_CLOSE);
62          
63          luoKyselyikkuna(kyselyikkuna.getContentPane());
64          
65          kyselyikkuna.pack();
66  //        kyselyikkuna.setVisible(true);
67      }
68      
69      private void luoKomponentit(Container container) {
70          BorderLayout layout = new BorderLayout(2, 1);
71          container.setLayout(layout);
72          
73          Piirtoalusta piirtoalusta = new Piirtoalusta(svl);
74          container.add(luoInfoikkuna(), BorderLayout.NORTH);
75          container.add(luoPalauteikkuna(), BorderLayout.SOUTH);
76          piirtoalusta.addMouseListener(new HiirenKuuntelija(guilg, piirtoalusta, siirrot, pelaaja, terveiset));
77          container.add(piirtoalusta);
78          guilg.lisaaPiirtoalusta(piirtoalusta);
79          
80      }
81      
82      private JPanel luoInfoikkuna() {
83          JPanel panel = new JPanel(new GridLayout(1, 2));
84          pelaaja = new JLabel(svl.getVuorossaOlevaPelaaja().toString());
85          siirrot = new JLabel("Siirrot: " + svl.haeSiirrot().toString());
86          panel.add(pelaaja);
87          panel.add(siirrot);
88          
89          return panel;
90      }
91      
92      private JPanel luoPalauteikkuna() {
93          JPanel panel = new JPanel(new GridLayout(1, 1));
94          terveiset = new JLabel("Palautetta toiminnasta t√§nne.");
95          panel.add(terveiset);
96          
97          return panel;
98      }
99      
100     private void luoKyselyikkuna(Container container) {
101         GridLayout layout = new GridLayout(3, 1);
102         container.setLayout(layout);
103         
104         JLabel kysymysTeksti = new JLabel("Montako pelaajaa? (0-2)");
105         JTextField vastausKentta = new JTextField();
106         JButton vastaaNappi = new JButton("OK");
107         container.add(kysymysTeksti);
108         container.add(vastausKentta);
109         container.add(vastaaNappi);
110         
111         KyselyKuuntelija kyselykuuntelija = new KyselyKuuntelija(kyselyikkuna, guilg, kysymysTeksti, vastausKentta);
112         vastaaNappi.addActionListener(kyselykuuntelija);
113     }
114     
115     private void luoPelilaudanKomponentit(Container container) {
116         container.add(new Piirtoalusta(svl));
117     }
118     
119     public JFrame getFrame() {
120         return kehys;
121     }
122     
123 }
